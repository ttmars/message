8.for range表达式

Fri, May 6, 2022

range闭包

每次迭代，i、v都是同一个变量

package main

import (
	"fmt"
	"time"
)

func main() {
	var s = []int{1,2,3,4,5}
	for i,v := range s {
		go func() {
			time.Sleep(time.Second*1)
			fmt.Println(i,v)
		}()
	}
	time.Sleep(time.Second*3)
}

//4 5
//4 5
//4 5
//4 5
//4 5

range数组、切片

package main

import "fmt"

func main() {
	var arr = [5]int{1,2,3,4,5}
	var arr1 [5]int
	for i,v := range arr {		// 理解此处range迭代的arr是arr的一个副本！
		if i == 0 {
			arr[1] = 11
			arr[2] = 12
		}
		
		arr1[i] = v
	}

	fmt.Println(arr)
	fmt.Println(arr1)
}

//[1 11 12 4 5]
//[1 2 3 4 5]

package main

import "fmt"

func main() {
	var arr = [5]int{1,2,3,4,5}
	var arr1 [5]int
	for i,v := range &arr {		// 通过传递指针，实现对原始数组的迭代！
		if i == 0 {
			arr[1] = 11
			arr[2] = 12
		}

		arr1[i] = v
	}

	fmt.Println(arr)
	fmt.Println(arr1)
}

//[1 11 12 4 5]
//[1 11 12 4 5]

range&string

range迭代string时，以rune类型迭代，而不是byte

range&map

package main

import "fmt"

func main() {
	var m = map[string]int{
		"Bob": 11,
		"Tom": 12,
		"Jim": 13,
	}
	count := 0
	for k,v := range m {
		if count == 0 {
			delete(m, "Jim")
		}
		count ++
		fmt.Println(k,v)
	}
	fmt.Println("count:", count)
}

// 结果不确定，若首次迭代到Jim，则count=3；否则count=2

range&channel

range迭代channel时，会逐步取出通道中的元素，然后阻塞直至通道关闭；若没有通道关闭代码，则会造成死锁

package main

import (
	"fmt"
	"time"
)

func main() {
	var ch = make(chan int, 3)
	ch <- 1
	ch <- 2
	ch <- 3
	go func() {
		time.Sleep(time.Second*3)
		close(ch)
	}()
	fmt.Println(len(ch))
	for v := range ch {
		fmt.Println(v, len(ch))
	}
}

//3
//1 2
//2 1
//3 0

package main

import "fmt"
