链表翻转

package main

import "fmt"

type Node struct {
	Data int
	Next *Node
}

// CreateList 通过切片初始化链表
func CreateList(data []int, head *Node)  {
	p := head
	for _,v := range data {
		p.Next = &Node{v, nil}
		p = p.Next
	}

}

// PrintList 打印链表
func PrintList(head *Node)  {
	p := head.Next
	for p!=nil {
		fmt.Printf("%d ", p.Data)
		p = p.Next
	}
	fmt.Println()
}

func main() {
	data := []int{3,8,1,9,6,2,7,4,5}
	//data := []int{3,8}
	//data := []int{3}
	head := &Node{}
	CreateList(data, head)
	PrintList(head)

	ReverseLink(head)
	PrintList(head)
}

// ReverseLink 链表逆序
func ReverseLink(head *Node)  {
	if head == nil || head.Next.Next == nil {		// 若链表为空，或者只有一个元素，则直接返回
		return
	}
	p1 := head.Next		// p1指向第一个节点
	p2 := p1.Next		// p2指向第二个节点
	p1.Next = nil		// 翻转后，第一个节点变成最后一个节点，将第一个节点的指针置空;否则会造成链表循环
	for {
		t := p2.Next	// 临时保存p2的下一个节点地址，防止丢失
		p2.Next = p1	// 翻转，将节点2的指针指向前一个节点
		p1 = p2			// 往后移动p1
		if t != nil {	// 如果p2节点的指针域不为nil，往后移动p2;否则p2已经是最后一个节点了，将头结点的指针域指向最后一个节点，并退出循环
			p2 = t
		}else{
			head.Next = p2
			break
		}
	}
}

//3 8 1 9 6 2 7 4 5
//5 4 7 2 6 9 1 8 3

删除链表中的重复元素

package main

import "fmt"

type Node struct {
	Data int
	Next *Node
}

// CreateList 通过切片初始化链表
func CreateList(data []int, head *Node)  {
	p := head
	for _,v := range data {
		p.Next = &Node{v, nil}
		p = p.Next
	}

}

// PrintList 打印链表
func PrintList(head *Node)  {
	p := head.Next
	for p!=nil {
		fmt.Printf("%d ", p.Data)
		p = p.Next
	}
	fmt.Println()
}

func main() {
	data := []int{1, 3, 1, 5 ,5, 7, 7}
	//data := []int{1,1}
	//data := []int{1}
	head := &Node{}
	CreateList(data, head)
	PrintList(head)

	DeleteDuplicates(head)
	PrintList(head)
}

// DeleteDuplicates 删除重复元素
func DeleteDuplicates(head *Node)  {
	outPtr := head.Next			// 外层循环遍历链表取值，内层循环遍历后边的元素与之比较，相同则删除
	for outPtr!=nil && outPtr.Next!=nil {
		p1 := outPtr			// 涉及到链表删除，需要记录当前节点的前一个节点，故通过双指针遍历
		p2 := p1.Next
		for {
			if p2.Data == outPtr.Data{
				if p2.Next == nil {		// 处理尾节点
					p1.Next = nil
					break
				}
				p1.Next = p2.Next
				p2 = p2.Next
			}else{
				if p2.Next == nil {		// 处理尾节点
					break
				}
				p1 = p1.Next
				p2 = p1.Next
			}
		}
		outPtr = outPtr.Next
	}
}

//1 3 1 5 5 7 7
//1 3 5 7

查找链表倒数第k个元素

package main

import "fmt"

type Node struct {
	Data int
	Next *Node
}

// CreateList 通过切片初始化链表
func CreateList(data []int, head *Node)  {
	p := head
	for _,v := range data {
		p.Next = &Node{v, nil}
		p = p.Next
	}

}

// PrintList 打印链表
func PrintList(head *Node)  {
	p := head.Next
	for p!=nil {
		fmt.Printf("%d ", p.Data)
		p = p.Next
	}
	fmt.Println()
}

func main() {
	data := []int{3,8,1,9,6,2,7,4,5}
	head := &Node{}
	CreateList(data, head)
	PrintList(head)

	fmt.Println(FindLastK(head, 1))
	fmt.Println(FindLastK(head, 3))
	fmt.Println(FindLastK(head, 9))
	fmt.Println(FindLastK(head, 900))
}

// FindLastK 快慢指针，查找链表倒数第k个元素
func FindLastK(head *Node, k int) int {
	pre := head.Next
	cur := pre
	for ;k>0;k-- {
		cur = cur.Next
		if cur == nil {
			return pre.Data
			break
		}
	}
	for cur.Next != nil {
		pre = pre.Next
		cur = cur.Next
	}
	return pre.Next.Data
}

//3 8 1 9 6 2 7 4 5
//5
//7
//3
//3

翻转链表相邻元素

package main

import "fmt"

type Node struct {
	Data int
	Next *Node
}

// CreateList 通过切片初始化链表
func CreateList(data []int, head *Node)  {
	p := head
	for _,v := range data {
		p.Next = &Node{v, nil}
		p = p.Next
	}

}

// PrintList 打印链表
func PrintList(head *Node)  {
	p := head.Next
	for p!=nil {
		fmt.Printf("%d ", p.Data)
		p = p.Next
	}
	fmt.Println()
}

func main() {
	data := []int{3,8,1,9,6,2,7,4,5}
	//data := []int{3,8}
	//data := []int{3}
	head := &Node{}
	CreateList(data, head)
	PrintList(head)

	Reverse(head)
	PrintList(head)
}

// Reverse 翻转链表的相邻元素
func Reverse(head *Node)  {
	p1 := head
	if p1.Next==nil || p1.Next.Next==nil{
		return
	}
	p2 := p1.Next
	p3 := p2.Next
	for {
		t := p3.Next
		p1.Next = p3
		p3.Next = p2
		p2.Next = t

		p1 = p2
		if p1.Next==nil || p1.Next.Next==nil{
			break
		}
		p2 = p1.Next
		p3 = p2.Next
	}
}

//3 8 1 9 6 2 7 4 5
//8 3 9 1 2 6 4 7 5
